**[Converted Go Code]**:
```go
// Converted Go functions
package main

import (
	"fmt"
	"math"
)

// AnalyzeData analyzes numerical data and returns mean and standard deviation.
func AnalyzeData(df [][]float64) (map[string]float64, string) {
	// Check for empty DataFrame
	if len(df) == 0 {
		return nil, "Error: Empty DataFrame"
	}

	// Compute mean and standard deviation
	var sum float64
	values := make([]float64, len(df))

	for i, row := range df {
		values[i] = row[0]
		sum += row[0]
	}

	mean := sum / float64(len(values))

	var variance float64
	for _, val := range values {
		variance += math.Pow(val-mean, 2)
	}
	variance /= float64(len(values))
	stdDev := math.Sqrt(variance)

	return map[string]float64{
		"mean":   mean,
		"stdDev": stdDev,
	}, ""
}
```
